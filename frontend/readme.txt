// to initialize a new project with default settings
npm init -y

// to install Tailwind CSS and the Vite plugin for Tailwind CSS
npm i tailwindcss @tailwindcss/vite

// to run the development server
npm run dev

// to format code with Prettier and Tailwind CSS plugin
npm i prettier prettier-plugin-tailwindcss

// to seed the database with initial data
php artisan db:seed
